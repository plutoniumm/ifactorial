    export const load = async ({ params }) => {
      let file = params.slug;
      const isJSON = file.includes('.json');

      if (isJSON) {
        file = file.replace('.json', '');
      }

      let books = await import(`../../../data/${file}.csv`);
      books = books.default;

      if (!books) {
        return { status: 404 };
      };

      const data =  {
        books,
        index: params.slug.match(/d/g).join(""),
      }

      return data;
    }
